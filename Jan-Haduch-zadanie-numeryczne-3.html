λ = -(1/10) + 1j # stała λ

# funkcja metody bisekcji
def bisect(f, arg1, arg2, e):
    while np.abs(arg1 - arg2) > e:
        if f(arg1) * f((arg1 + arg2)/2) < 0:
            arg2 = (arg1 + arg2)/2
        else:
            arg1 = (arg1 + arg2)/2
    return (arg1 + arg2)/2

R = lambda h: np.abs(1 + (λ * h) + ((λ * h)**2)/2) - 1

h_gran = bisect(R, 1/2, 2, 10**(-12))
print(f'Krok graniczny wynosi {h_gran}.')

Krok graniczny wynosi 1.0424613808173717.

Warto mieć w pamięci wzory definiujące metody Rungego-Kutny: $$ x_{k+1} = x_k + h\sum_{i=1}^{s}{b_i K_i},\quad K_i = f\bigg( t_k + c_i h,\space x_k + h\sum_{j=1}^{s}{a_{ij} K_j}\bigg),\quad i = 1, 2,\dots , s. $$

Schemat Heunego, użyty w tym zadaniu, ma postać: $$ M = \begin{bmatrix} 0 & 0 \\ 1 & 0 \end{bmatrix},\space b^T = \begin{bmatrix} \frac{1}{2} & \frac{1}{2} \end{bmatrix},\space c = \begin{bmatrix} 0 \\ 1 \end{bmatrix}. $$

Schemat ten pociąga wzór: $$ x_{k+1} = x_k + \frac{h}{2} f(t_k ,\space x_k) + \frac{h}{2} f\Big( t_k + h,\space x_k + hf(t_k ,x_k )\Big) $$

W naszym przypadku funkcja f to macierz $$ A = \begin{bmatrix} -\frac{1}{10} & 1 \\ -1 & -\frac{1}{10} \end{bmatrix}, $$ natomiast wartość początkowa to $$ x(0)=\begin{bmatrix} \frac{1}{2} \\ -\frac{1}{3} \end{bmatrix} $$

               [1, 0]])
x0 = np.array([1/2, -1/3]) # macierz warunków początkowych
b = np.array([1/2, 1/2]) # macierz b metody rungego kutny
c = np.array([0, 1]) # macierz c metody rungego kutny

A = np.array([[-1/10, 1], [-1, -1/10]]) # macierz prawej strony równania różniczkowego

f = lambda x: A@x # funkcja prawej strony równania różniczkowego

# f(x0)
K1 = lambda x: f(x) # wektor K1 metody RK

K2 = lambda x, h: f(x + h*K1(x)) # wektor K2 metody RK

nast = lambda xk, h: xk + h*(b[0]*K1(xk) + b[1]*K2(xk, h)) # funkcja zwracająca x_{k+1}
# nast = lambda xk, h: xk + (h/2)*f(xk) + (h/2)*f(xk + (h/2)*f(xk))

def rozwiazanie(h, i): # funkcja zwracająca listę wektorów przybliżeń
    xk = x0 
    rozw1 = [x0[0]] # lista wartości aproksymacji na pierwszej współrzędnej
    rozw2 = [x0[1]] # lista wartości aproksymacji na pierwszej współrzędnej
    for k in range(i-1):
        xk1 = nast(xk, h) # wyliczam następny wektor aproksymacji
        rozw1.append(xk1[0])
        rozw2.append(xk1[1])
        xk = xk1 # przygotowanie do następnego kroku
    return rozw1, rozw2
